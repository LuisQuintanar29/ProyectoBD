CREATE DATABASE papeleria;

\c papeleria

DROP TABLE IF EXISTS cliente CASCADE;
DROP TABLE IF EXISTS compañia CASCADE;
DROP TABLE IF EXISTS consumo CASCADE;
DROP TABLE IF EXISTS email CASCADE;
DROP TABLE IF EXISTS impresion CASCADE;
DROP TABLE IF EXISTS inventario CASCADE;
DROP TABLE IF EXISTS preciorecarga CASCADE;
DROP TABLE IF EXISTS producto CASCADE;
DROP TABLE IF EXISTS proveedor CASCADE;
DROP TABLE IF EXISTS recarga CASCADE;
DROP TABLE IF EXISTS tamañohoja CASCADE;
DROP TABLE IF EXISTS telefono CASCADE;
DROP TABLE IF EXISTS tipoimpresion CASCADE;
DROP TABLE IF EXISTS tipoproducto CASCADE;

CREATE TABLE cliente(
RFC varchar(13) NOT NULL,
NombreCliente varchar(70) NOT NULL ,
DomicilioCliente varchar(70) NOT NULL,
PRIMARY KEY (RFC)
);

CREATE TABLE email(
Email varchar(50) NOT NULL,
RFC varchar(13) NOT NULL,
PRIMARY KEY (Email),
FOREIGN KEY (RFC) REFERENCES cliente(RFC)
);

CREATE TABLE proveedor(
RazonSocial varchar(30) NOT NULL,
DomicilioProv varchar(30) NOT NULL,
NombreProv varchar(70) NOT NULL,
PRIMARY KEY (RazonSocial)
);

CREATE TABLE telefono(
Telefono int,
RazonSocial varchar(30) NOT NULL,
PRIMARY KEY (Telefono),
FOREIGN KEY (RazonSocial) REFERENCES proveedor (RazonSocial)
);

CREATE TABLE inventario(
CodigoBarras int,
PrecioCompra int,
FechaCompra date,
Stock int,
PRIMARY KEY (CodigoBarras)
);

CREATE TABLE tipoproducto(
IDTipoProd serial,
TipoProducto varchar(30) NOT NULL,
PRIMARY KEY (IDTipoProd)
);

CREATE TABLE producto(
CodigoBarras int,
PrecioProd int,
Marca varchar(15) NOT NULL,
DescripcionProd varchar(30) NOT NULL,
IDTipoProd int,
PRIMARY KEY (CodigoBarras),
FOREIGN KEY (CodigoBarras) REFERENCES inventario (CodigoBarras),
FOREIGN KEY (IDTipoProd) REFERENCES tipoproducto (IDTipoProd)
);

CREATE TABLE tipoimpresion(
IDTipoImpresion serial,
TipoImpresion varchar(15) NOT NULL,
PRIMARY KEY (IDTipoImpresion)
);

CREATE TABLE tamañohoja(
IDTamaño serial,
TamañoHoja varchar(15) NOT NULL,
PRIMARY KEY (IDTamaño)
);

CREATE TABLE impresion(
IDTamaño int,
IDTipoImpresion int,
PrecioServ int,
DescripcionServ varchar(30) NOT NULL,
PRIMARY KEY (IDTamaño,IDTipoImpresion),
FOREIGN KEY (IDTipoImpresion) REFERENCES tipoimpresion (IDTipoImpresion),
FOREIGN KEY (IDTamaño) REFERENCES tamañohoja (IDTamaño)
);

CREATE TABLE compañia(
IDCompañia serial,
Compañia varchar(30) NOT NULL,
PRIMARY KEY (IDCompañia)
);

CREATE TABLE preciorecarga(
IDRecarga serial,
PrecioRecarga int,
PRIMARY KEY (IDRecarga)
);

CREATE TABLE recarga(
IDCompañia int,
IDRecarga int,
DescripcionServ varchar(30) NOT NULL,
PRIMARY KEY (IDCompañia,IDRecarga),
FOREIGN KEY (IDCompañia) REFERENCES compañia (IDCompañia),
FOREIGN KEY (IDRecarga) REFERENCES  preciorecarga (IDRecarga)
);

CREATE TABLE consumo(
NoVenta int,
FechaVenta date,
CantidadArticulo int,
PrecioArticulo int,
Total int GENERATED ALWAYS AS (CantidadArticulo*PrecioArticulo) STORED
);
